<script>
let ws;
let myKey;
let username;
let currentChatKey = null;
let friends = {};

function logMessage(from, content, self=false) {
const chatBox = document.getElementById("chatBox");
const div = document.createElement("div");
div.className = "message " + (self ? "self" : "other");
div.textContent = from + ": " + content;
chatBox.appendChild(div);
chatBox.scrollTop = chatBox.scrollHeight;
}

function connect() {
myKey = document.getElementById("myKey").value.trim();
username = document.getElementById("username").value.trim();
if(!myKey || !username) { alert("Enter both key and username"); return; }

ws = new WebSocket("ws://localhost:8765"); // Replace with your server
ws.onopen = () => {
ws.send(JSON.stringify({type:"connect", friend_key:myKey, username:username}));
alert("Connected!");
};
ws.onmessage = (event) => {
const data = JSON.parse(event.data);
if(data.type === "message") {
logMessage(data.from, data.message, false);
} else if(data.type === "friend_request") {
addFriendRequest(data.from);
} else if(data.type === "friend_accept") {
addFriend(data.from, data.username);
}
};
}

function sendMessage() {
const msg = document.getElementById("messageInput").value.trim();
if(!msg || !currentChatKey) return;
ws.send(JSON.stringify({type:"message", to:currentChatKey, message:msg}));
logMessage(username, msg, true);
document.getElementById("messageInput").value = "";
}

function sendFriendRequest() {
const key = document.getElementById("addFriendKey").value.trim();
if(!key) return;
ws.send(JSON.stringify({type:"friend_request", to:key}));
alert("Friend request sent to " + key);
document.getElementById("addFriendKey").value = "";
}

function addFriendRequest(key) {
const div = document.createElement("div");
div.className = "request";
div.textContent = "Friend request from " + key + " (click to accept)";
div.onclick = () => {
ws.send(JSON.stringify({type:"friend_accept", to:key}));
addFriend(key, key);
div.remove();
};
document.getElementById("requests").appendChild(div);
}

function addFriend(key, name) {
friends[key] = name;
const div = document.createElement("div");
div.className = "friend";
div.textContent = name + " (" + key + ")";
div.onclick = () => { currentChatKey = key; document.getElementById("chatBox").innerHTML=""; };
document.getElementById("friends").appendChild(div);
}
